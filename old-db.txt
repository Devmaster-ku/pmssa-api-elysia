    public function canManageInFaculty($departmentId): bool
    {
        // Super Admin เข้าถึงได้ทุกอย่าง
        if ($this->isSuperAdmin()) {
            return true;
        }

        // Admin ธรรมดา - ตรวจสอบตามคณะ

    /**
     * ดึงรายการ ID ของหน่วยงานทั้งหมดที่ Admin สามารถจัดการได้ในคณะของตน
     */
    public function getManageableDepartmentIdsInFaculty(): array
    {
        // Super Admin เห็นทุกหน่วยงาน
        if ($this->isSuperAdmin()) {
            try {
                return Department::where('is_active', true)->pluck('id')->toArray();
            } catch (\Exception $e) {
                Log::warning('Failed to get all departments for super admin', [
                    'user_id' => $this->id,
                    'error' => $e->getMessage()
                ]);
                return [];
            }
        }

        // Admin/Manager - เฉพาะในคณะของตนเอง
        if (!$this->isAdmin() && !$this->isManager()) {
            return $this->getAccessibleDepartmentIds()->toArray();
        }

        try {
            $manageableIds = collect();
            
            // ดึงหน่วยงานที่ admin นี้สังกัด
            $userDepartments = $this->departments()->get();
            
            foreach ($userDepartments as $userDept) {
                // หาคณะหลักของหน่วยงานนี้
                $faculty = $this->getFacultyDepartment($userDept);
                
                if ($faculty) {
                    // เพิ่มคณะหลัก
                    $manageableIds->push($faculty->id);
                    
                    // เพิ่มหน่วยงานย่อยทั้งหมดในคณะนี้
                    $subDepartments = $faculty->getAllDescendants();
                    foreach ($subDepartments as $sub) {
                        $manageableIds->push($sub->id);
                    }
                }
            }

            return $manageableIds->unique()->values()->toArray();
            
        } catch (\Exception $e) {
            \Illuminate\Support\Facades\Log::warning('Failed to get manageable department IDs for admin', [
                'user_id' => $this->id,
                'error' => $e->getMessage()
            ]);
            
            // Fallback to accessible departments
            return $this->getAccessibleDepartmentIds()->toArray();
        }
    }

    /**
     * ดึงหน่วยงานหลักของผู้ใช้
     */
    public function getPrimaryDepartment(): ?Department
    {
        try {
            // วิธีที่ 1: ดึงจาก pivot table (primary method)
            $departmentUser = $this->activeDepartmentUsers()
                ->with('department')
                ->orderByDesc('role') // Admin, Manager มาก่อน
                ->orderByDesc('created_at') // ล่าสุดมาก่อน
                ->first();

            if ($departmentUser && $departmentUser->department) {
                return $departmentUser->department;
            }

            // วิธีที่ 2: ดึงจาก direct foreign key
            if (isset($this->attributes['department_id']) && $this->department_id) {
                return Department::find($this->department_id);
            }

            // วิธีที่ 3: ดึงจาก department field (legacy)
            if (isset($this->attributes['department']) && $this->department) {
                return Department::where('name', $this->department)
                    ->where('is_active', true)
                    ->first();
            }

        } catch (\Exception $e) {
            Log::warning('Failed to get primary department', [
                'user_id' => $this->id,
                'error' => $e->getMessage()
            ]);
        }

        return null;
    }

    public function getUserDepartmentId()
    {
        // Check if user has department relationship
        $departmentUser = $this->departmentUsers()->where('is_active', true)->first();

    /**
     * 🔥 ดึงหน่วยงานหลัก (คณะ) ทั้งหมดที่ผู้ใช้สามารถเข้าถึงได้
     * รองรับทั้งแบบ many-to-many และ direct assignment
     */
    public function getAccessibleFaculties()
    {
        try {
            // Super Admin เข้าถึงได้ทุกคณะ
            if ($this->isSuperAdmin()) {
                return Department::where('type', 'main')
                    ->where('is_active', true)
                    ->orderBy('name')
                    ->get();
            }

            $faculties = collect();

            // วิธีที่ 1: ดึงจาก many-to-many relationship
            if (method_exists($this, 'departments')) {
                $userDepartments = $this->departments()->get();
                
                foreach ($userDepartments as $dept) {
                    $faculty = $this->findUserFacultyDepartment($dept);
                    if ($faculty) {
                        $faculties->push($faculty);
                    }
                }
            }

            // วิธีที่ 2: ดึงจาก department field (เก่า)
            if ($faculties->isEmpty() && !empty($this->department)) {
                $userDept = Department::where('name', $this->department)
                    ->where('is_active', true)
                    ->first();
                    
                if ($userDept) {
                    $faculty = $this->findUserFacultyDepartment($userDept);
                    if ($faculty) {
                        $faculties->push($faculty);
                    }
                }
            }

            // วิธีที่ 3: ดูจากหน่วยงานที่สร้างโดย user
            if ($faculties->isEmpty()) {
                $createdDepts = Department::where('created_by', $this->id)
                    ->where('is_active', true)
                    ->get();
                    
                foreach ($createdDepts as $dept) {
                    $faculty = $this->findUserFacultyDepartment($dept);
                    if ($faculty) {
                        $faculties->push($faculty);
                    }
                }
            }

            return $faculties->unique('id')->sortBy('name');

        } catch (\Exception $e) {
            Log::error('Failed to get accessible faculties', [
                'user_id' => $this->id,
                'error' => $e->getMessage()
            ]);
            
            return collect();
        }
    }

    /**
     * 🔥 หาหน่วยงานหลัก (คณะ) จากหน่วยงานที่กำหนด
     * ปรับปรุงจาก getFacultyDepartment() เดิม
     */
    public function findUserFacultyDepartment($department)
    {
        if (!$department) {
            return null;
        }

        try {
            // ถ้าเป็นหน่วยงานหลักอยู่แล้ว
            if ($department->type === 'main') {
                return $department;
            }

            // หาขึ้นไปจนถึงหน่วยงานหลัก
            $current = $department;
            $maxDepth = 10; // ป้องกัน infinite loop
            $depth = 0;
            $visited = collect([$department->id]);

            while ($current && $current->parent_id && $depth < $maxDepth) {
                // ป้องกัน circular reference
                if ($visited->contains($current->parent_id)) {
                    Log::warning('Circular reference detected in department hierarchy', [
                        'user_id' => $this->id,
                        'department_id' => $department->id,
                        'current_id' => $current->id,
                        'parent_id' => $current->parent_id
                    ]);
                    break;
                }

                $parent = Department::find($current->parent_id);
                if (!$parent || !$parent->is_active) {
                    break;
                }

                $visited->push($parent->id);

                if ($parent->type === 'main') {
                    return $parent;
                }
                
                $current = $parent;
                $depth++;
            }

            // ถ้าไม่พบหน่วยงานหลัก อาจเป็น standalone department
            Log::info('No faculty found for department', [
                'user_id' => $this->id,
                'department_id' => $department->id,
                'department_name' => $department->name,
                'department_type' => $department->type
            ]);

        } catch (\Exception $e) {
            Log::error('Error finding user faculty department', [
                'user_id' => $this->id,
                'department_id' => $department->id ?? null,
                'error' => $e->getMessage()
            ]);
        }

        return null;
    }

    /**
     * 🔥 ดึงหน่วยงานทั้งหมดที่ผู้ใช้สามารถเข้าถึงได้ (สำหรับ filter)
     * รวมหน่วยงานหลักและหน่วยงานย่อย
     */
    public function getAccessibleDepartmentsForFilter()
    {
        try {
            // Super Admin เข้าถึงได้ทุกหน่วยงาน
            if ($this->isSuperAdmin()) {
                return Department::where('is_active', true)
                    ->orderBy('type', 'desc') // main ก่อน sub
                    ->orderBy('name')
                    ->get();
            }

            $allDepartments = collect();
            $faculties = $this->getAccessibleFaculties();

            foreach ($faculties as $faculty) {
                // เพิ่มหน่วยงานหลัก (คณะ)
                $allDepartments->push($faculty);
                
                // เพิ่มหน่วยงานย่อยทั้งหมดภายใต้คณะนี้
                $subDepartments = $this->getAllSubDepartmentsRecursive($faculty->id);
                $allDepartments = $allDepartments->merge($subDepartments);
            }

            return $allDepartments->unique('id')
                ->sortBy(['type', 'name']); // เรียงตาม type แล้วตาม name

        } catch (\Exception $e) {
            Log::error('Failed to get accessible departments for filter', [
                'user_id' => $this->id,
                'error' => $e->getMessage()
            ]);
            
            return collect();
        }
    }

    /**
     * 🔥 ดึงหน่วยงานย่อยทั้งหมดภายใต้หน่วยงานหลัก (recursive)
     */
    private function getAllSubDepartmentsRecursive($parentId)
    {
        $allSubs = collect();
        
        try {
            $directSubs = Department::where('parent_id', $parentId)
                ->where('is_active', true)
                ->orderBy('name')
                ->get();
            
            foreach ($directSubs as $sub) {
                $allSubs->push($sub);
                
                // Recursive สำหรับหน่วยงานย่อยระดับลึกกว่า
                $deeperSubs = $this->getAllSubDepartmentsRecursive($sub->id);
                $allSubs = $allSubs->merge($deeperSubs);
            }
            
        } catch (\Exception $e) {
            Log::error('Error getting sub-departments recursively', [
                'parent_id' => $parentId,
                'user_id' => $this->id,
                'error' => $e->getMessage()
            ]);
        }
        
        return $allSubs;
    }

    /**
     * 🔥 ตรวจสอบว่าผู้ใช้สามารถเข้าถึงหน่วยงานนี้ได้หรือไม่ (Enhanced)
     */
    public function canAccessDepartmentEnhanced($departmentId): bool
    {
        if (!$departmentId) {
            return false;
        }

        // Super Admin เข้าถึงได้ทุกหน่วยงาน
        if ($this->isSuperAdmin()) {
            return true;
        }

        try {
            $accessibleDepartmentIds = $this->getAccessibleDepartmentsForFilter()
                ->pluck('id')
                ->toArray();
                
            return in_array($departmentId, $accessibleDepartmentIds);

        } catch (\Exception $e) {
            Log::error('Error checking department access', [
                'user_id' => $this->id,
                'department_id' => $departmentId,
                'error' => $e->getMessage()
            ]);
            
            return false;
        }
    }

    /**
     * ดึงรายการหน่วยงานทั้งหมดที่ admin สามารถจัดการได้
     */
    public function getManageableDepartmentIds(): array
    {
        if ($this->isSuperAdmin()) {
            return Department::where('is_active', true)->pluck('id')->toArray();
        }

        if (!$this->isAdmin()) {
            // Non-admin users ใช้วิธีเดิม
            return $this->getAccessibleDepartmentIds()->toArray();
        }

        // Department Admin - เข้าถึงได้ทั้งคณะ
        $manageableIds = collect();
        
        foreach ($this->departments as $dept) {
            $faculty = $this->getFacultyDepartment($dept);
            if ($faculty) {
                // เพิ่มคณะหลัก
                $manageableIds->push($faculty->id);
                
                // เพิ่มหน่วยงานย่อยทั้งหมดในคณะ
                $subDepartments = $this->getAllSubDepartments($faculty->id);
                $manageableIds = $manageableIds->merge($subDepartments);
            }
        }

        return $manageableIds->unique()->values()->toArray();
    }

    /**
     * ดึงหน่วยงานย่อยทั้งหมดภายในคณะ
     */
    private function getAllSubDepartments($facultyId): \Illuminate\Support\Collection
    {
        $allSubs = collect();
        
        $directSubs = Department::where('parent_id', $facultyId)
            ->where('is_active', true)
            ->get();
        
        foreach ($directSubs as $sub) {
            $allSubs->push($sub->id);
            // Recursive สำหรับหน่วยงานย่อยระดับลึกกว่า
            $deeperSubs = $this->getAllSubDepartments($sub->id);
            $allSubs = $allSubs->merge($deeperSubs);
        }
        
        return $allSubs;
    }

    /**
     * ตรวจสอบว่าผู้ใช้สามารถเข้าถึงโครงการได้หรือไม่
     * รองรับทั้งแบบ direct department_id และ many-to-many relationship
     * 
     * @param Project $project โครงการที่ต้องการตรวจสอบ
     * @return bool
     */
    public function canAccessProject($project): bool
    {
        // ตรวจสอบว่า $project เป็น object ที่มี properties ที่จำเป็น
        if (!$project || !isset($project->department_id)) {
            return false;
        }

        // Super Admin สามารถเข้าถึงได้ทุกโครงการ
        if ($this->isSuperAdmin()) {
            return true;
        }

        // Admin สามารถเข้าถึงได้ทุกโครงการในคณะของตน
        if ($this->isAdmin()) {
            return $this->canAccessFacultyData($project->department_id);
        }

        // เจ้าของโครงการ
        if (isset($project->created_by) && $project->created_by === $this->id) {
            return true;
        }

        // ตรวจสอบการเป็นสมาชิกในหน่วยงานเดียวกัน
        return $this->belongsToDepartmentEnhanced($project->department_id);
    }

    /**
     * ตรวจสอบการเป็นสมาชิกของหน่วยงาน (Enhanced version)
     * รองรับทั้ง many-to-many และ direct foreign key
     * 
     * @param int|string $departmentId ID ของหน่วยงาน
     * @return bool
     */
    public function belongsToDepartmentEnhanced($departmentId): bool
    {
        if (!$departmentId) {
            return false;
        }

        try {
            // วิธีที่ 1: ตรวจสอบผ่าน pivot table (primary method)
            if (method_exists($this, 'departmentUsers') && $this->departmentUsers()->exists()) {
                $exists = $this->departmentUsers()
                    ->where('department_id', $departmentId)
                    ->where('is_active', true)
                    ->exists();
                
                if ($exists) {
                    return true;
                }
            }

            // วิธีที่ 2: ตรวจสอบผ่าน direct foreign key
            if (isset($this->attributes['department_id']) && $this->department_id) {
                if ((int) $this->department_id === (int) $departmentId) {
                    return true;
                }
            }

            // วิธีที่ 3: ตรวจสอบผ่าน department field (string comparison)
            if (isset($this->attributes['department']) && $this->department) {
                // ถ้า $departmentId เป็น string ให้เปรียบเทียบตรงๆ
                if (is_string($departmentId)) {
                    return $this->department === $departmentId;
                }
                
                // ถ้า $departmentId เป็น int ให้ดึงชื่อจาก Department Model
                try {
                    $department = Department::find($departmentId);
                    return $department && $this->department === $department->name;
                } catch (\Exception $e) {
                    // ถ้าไม่มี Department Model หรือเกิดข้อผิดพลาด
                    return false;
                }
            }

        } catch (\Exception $e) {
            \Illuminate\Support\Facades\Log::warning('Failed to check department membership', [
                'user_id' => $this->id,
                'department_id' => $departmentId,
                'error' => $e->getMessage()
            ]);
        }

        return false;
    }

    /**
     * ดึงสถิติโครงการที่ผู้ใช้สามารถเข้าถึงได้
     * 
     * @param int $days จำนวนวันย้อนหลัง
     * @param bool $onlyUserProjects แสดงเฉพาะโครงการของผู้ใช้
     * @return array
     */
    public function getAccessibleProjectStats(int $days = 30, bool $onlyUserProjects = false): array
    {
        try {
            $startDate = now()->subDays($days);
            
            // ดึง accessible departments
            $accessibleDepartmentIds = $this->getAccessibleDepartmentIds()->toArray();
            
            // สร้าง base query
            $query = Project::where('created_at', '>=', $startDate);
            
            if ($this->isSuperAdmin()) {
                // Super Admin เห็นทุกอย่าง - ไม่ต้องกรอง
            } elseif ($onlyUserProjects) {
                // แสดงเฉพาะโครงการของผู้ใช้
                $query->where('created_by', $this->id);
            } else {
                // กรองตาม accessible departments และโครงการของตนเอง
                $query->where(function($q) use ($accessibleDepartmentIds) {
                    $q->where('created_by', $this->id);
                    
                    if (!empty($accessibleDepartmentIds)) {
                        $q->orWhereIn('department_id', $accessibleDepartmentIds);
                    }
                });
            }

            // นับตามสถานะ
            $stats = [
                'total' => $query->count(),
                'draft' => $query->clone()->where('status', 'draft')->count(),
                'pending_approval' => $query->clone()->where('status', 'pending_approval')->count(),
                'approved' => $query->clone()->where('status', 'approved')->count(),
                'rejected' => $query->clone()->where('status', 'rejected')->count(),
                'active' => $query->clone()->where('status', 'active')->count(),
                'completed' => $query->clone()->where('status', 'completed')->count(),
                'cancelled' => $query->clone()->where('status', 'cancelled')->count(),
            ];

            // คำนวณ derived stats
            $stats['my_projects'] = Project::where('created_by', $this->id)
                ->where('created_at', '>=', $startDate)
                ->count();
                
            $submittedForApproval = $stats['approved'] + $stats['rejected'];
            $stats['approval_rate'] = $submittedForApproval > 0 
                ? round(($stats['approved'] / $submittedForApproval) * 100, 1)
                : 0;

            return $stats;
            
        } catch (\Exception $e) {
            Log::error('Failed to get accessible project stats', [
                'user_id' => $this->id,
                'days' => $days,
                'only_user_projects' => $onlyUserProjects,
                'error' => $e->getMessage()
            ]);
            
            // Return empty stats on error
            return [
                'total' => 0,
                'draft' => 0,
                'pending_approval' => 0,
                'approved' => 0,
                'rejected' => 0,
                'active' => 0,
                'completed' => 0,
                'cancelled' => 0,
                'my_projects' => 0,
                'approval_rate' => 0,
            ];
        }
    }

    /**
     * ตรวจสอบว่าผู้ใช้สามารถจัดการเอกสารได้หรือไม่
     * 
     * @param ProjectDocument $document เอกสารที่ต้องการจัดการ
     * @return bool
     */
    public function canManageDocument($document): bool
    {
        // Admin สามารถจัดการได้ทุกเอกสาร
        if ($this->isAnyAdmin()) {
            return true;
        }

        // ผู้อัปโหลดสามารถจัดการเอกสารของตนเอง
        if ($document->uploaded_by === $this->id) {
            return true;
        }

        // ผู้ที่สามารถเข้าถึงโครงการสามารถจัดการเอกสารได้
        if ($this->canAccessProject($document->project)) {
            $role = $this->getRoleInDepartment($document->project->department_id);
            // เฉพาะ admin, manager, project-manager เท่านั้นที่จัดการเอกสารของคนอื่นได้
            return in_array($role, ['admin', 'manager', 'project-manager']);
        }

        return false;
    }

    /**
     * ตรวจสอบว่าผู้ใช้สามารถอนุมัติเอกสารได้หรือไม่
     * 
     * @param ProjectDocument $document เอกสารที่ต้องการอนุมัติ
     * @return bool
     */
    public function canApproveDocument($document): bool
    {
        // Super Admin สามารถอนุมัติได้ทุกเอกสาร
        if ($this->isSuperAdmin()) {
            return true;
        }

        // ตรวจสอบสิทธิ์ระดับหน่วยงาน
        $role = $this->getRoleInDepartment($document->project->department_id);
        return in_array($role, ['admin', 'manager', 'project-manager']);
    }

    /**
     * ดึงโครงการที่ผู้ใช้สามารถเข้าถึงได้
     * 
     * @return \Illuminate\Database\Eloquent\Builder
     */
    public function accessibleProjects()
    {
        if ($this->isAnyAdmin()) {
            // Admin เห็นทุกโครงการ
            return \App\Models\Project::query();
        }

        $accessibleDepartmentIds = $this->getAccessibleDepartmentIds();

        // โครงการที่ผู้ใช้สร้าง หรือ อยู่ในหน่วยงานที่เข้าถึงได้
        return \App\Models\Project::where(function($query) use ($accessibleDepartmentIds) {
            $query->where('created_by', $this->id);
            
            if ($accessibleDepartmentIds->isNotEmpty()) {
                $query->orWhereIn('department_id', $accessibleDepartmentIds->toArray());
            }
        });
    }

            $updateData['name'] = $name;
        }

        if ($department && $department !== $this->department) {
            $updateData['department'] = self::mapKUDepartmentToOption($department);
        }

        // Update KU specific data
        if (isset($userInfo['uid']) && $userInfo['uid'] !== $this->ku_uid) {
            $updateData['ku_uid'] = $userInfo['uid'];
        }
        if (isset($userInfo['faculty-id']) && $userInfo['faculty-id'] !== $this->ku_faculty_id) {
            $updateData['ku_faculty_id'] = $userInfo['faculty-id'];
        }
        if (isset($userInfo['position']) && $userInfo['position'] !== $this->ku_position) {
            $updateData['ku_position'] = $userInfo['position'];
        }

        if (!empty($updateData)) {
            $this->update($updateData);
        }
    }

    /**
     * ดึง roles ทั้งหมดจากทุก DepartmentUser ที่ active
     * รวม roles หลายค่า + role เดี่ยว
     * @return array
     */
    public function getAllUserRoles(): array
    {
        $allRoles = collect();

        // เพิ่ม role หลักจาก User table
        if (!empty($this->role)) {
            $allRoles->push($this->role);
        }

        // ดึง roles จากทุก active DepartmentUser
        try {
            $departmentUsers = $this->activeDepartmentUsers()->get(['role', 'roles']);
            
            foreach ($departmentUsers as $du) {
                // เพิ่ม role เดี่ยว
                if (!empty($du->role)) {
                    $allRoles->push($du->role);
                }
                // เพิ่ม roles หลายค่า
                if (!empty($du->roles) && is_array($du->roles)) {
                    foreach ($du->roles as $role) {
                        $allRoles->push($role);
                    }
                }
            }
        } catch (\Exception $e) {
            Log::warning('Failed to get department user roles', [
                'user_id' => $this->id,
                'error' => $e->getMessage()
            ]);
        }

        return $allRoles->unique()->values()->toArray();
    }

    /**
     * ตรวจสอบว่า user มี role ใดๆ ในรายการหรือไม่
     * ใช้สำหรับการตรวจสอบสิทธิ์เข้าถึงเมนู/หน้าต่างๆ
     * @param array $roles
     * @return bool

    public function hasAnyUserRole(array $roles): bool
    {

        return !empty(array_intersect($roles, $userRoles));
    }

    /**
     * ตรวจสอบว่า user มี role นี้หรือไม่ (จากทุกแหล่ง)

    /**
     * ดึง role หลักที่มี priority สูงสุด (จากทุกแหล่ง)
     * @return string
     */
    public function getPrimaryUserRole(): string
    {
        $allRoles = $this->getAllUserRoles();

        if (empty($allRoles)) {
            return $this->role ?? 'member';
        }

        $highestRole = 'member';
        $highestPriority = PHP_INT_MAX;

        foreach ($allRoles as $role) {
            $priority = DepartmentUser::ROLE_PRIORITY[$role] ?? 999;
            if ($priority < $highestPriority) {
                $highestPriority = $priority;
                $highestRole = $role;
            }
        }

        return $highestRole;
    }

    /**
     * ดึงชื่อแสดงของบทบาททั้งหมด (รองรับหลาย roles)
     * แปลง role keys เป็นชื่อภาษาไทยที่อ่านได้
     * @return string
     */
    public function getAllUserRolesDisplay(): string
    {
        $allRoles = $this->getAllUserRoles();
        
        if (empty($allRoles)) {
            return DepartmentUser::ROLES['member'] ?? 'สมาชิก';
        }

        $roleNames = array_map(function ($role) {
            return DepartmentUser::ROLES[$role] ?? $role;
        }, $allRoles);

        return implode(', ', $roleNames);
    }

    /**
 * ปรับปรุง getUserDepartment() ให้แม่นยำมากขึ้น
 */
public function getUserDepartment()
{
    try {
        // วิธีที่ 1: หาจาก DepartmentUser relation (ลำดับความสำคัญสูงสุด)
        $departmentUser = $this->activeDepartmentUsers()
            ->with('department')
            ->orderByRaw("CASE 
                WHEN role = 'admin' THEN 1 
                WHEN role = 'manager' THEN 2 
                WHEN role = 'project-manager' THEN 3 
                ELSE 4 END")
            ->orderByDesc('created_at')
            ->first();

        if ($departmentUser && $departmentUser->department && $departmentUser->department->is_active) {
            return $departmentUser->department;
        }

        // วิธีที่ 2: หาจาก department field ใน users table
        if (!empty($this->department)) {
            $department = Department::where('name', $this->department)
                ->where('is_active', true)
                ->first();
                
            if ($department) {
                return $department;
            }
        }

        // วิธีที่ 3: หาจากการที่สร้างหน่วยงาน
        $createdDepartment = Department::where('created_by', $this->id)
            ->where('is_active', true)
            ->orderByDesc('created_at')
            ->first();

        return $createdDepartment;

    } catch (\Exception $e) {
        Log::warning('Failed to get user department', [
            'user_id' => $this->id,
            'error' => $e->getMessage()
        ]);
        
        return null;
    }
}

/**
 * ปรับปรุง getFacultyDepartment() ให้มี error handling และป้องกัน infinite loop
 */
public function getFacultyDepartment($department)
{
    if (!$department) {
        return null;
    }

    try {
        // ถ้าเป็นหน่วยงานหลักอยู่แล้ว
        if ($department->type === 'main') {
            return $department;
        }

        // หาหน่วยงานแม่จนถึงหน่วยงานหลัก
        $current = $department;
        $maxDepth = 10; // ป้องกัน infinite loop
        $depth = 0;
        $visited = collect([$department->id]); // ป้องกัน circular reference

        while ($current && $current->parent_id && $depth < $maxDepth) {
            // ตรวจสอบ circular reference
            if ($visited->contains($current->parent_id)) {
                Log::warning('Circular reference detected in department hierarchy', [
                    'user_id' => $this->id,
                    'department_id' => $department->id,
                    'current_id' => $current->id,
                    'parent_id' => $current->parent_id,
                    'visited' => $visited->toArray()
                ]);
                break;
            }

            $parent = Department::find($current->parent_id);
            if (!$parent || !$parent->is_active) {
                break;
            }

            $visited->push($parent->id);

            if ($parent->type === 'main') {
                return $parent;
            }
            
            $current = $parent;
            $depth++;
        }

        Log::info('No faculty department found', [
            'user_id' => $this->id,
            'original_department_id' => $department->id,
            'depth_reached' => $depth
        ]);

    } catch (\Exception $e) {
        Log::error('Error finding faculty department', [
            'user_id' => $this->id,
            'department_id' => $department->id ?? null,
            'error' => $e->getMessage()
        ]);
    }

    return null;
}

/**
 * Helper method สำหรับหาหน่วยงานย่อยทั้งหมด (recursive)
 */
private function getAllSubDepartmentIds($facultyId): \Illuminate\Support\Collection
{
    $allIds = collect();
    
    try {
        $directSubs = Department::where('parent_id', $facultyId)
            ->where('is_active', true)
            ->get();
        
        foreach ($directSubs as $sub) {
            $allIds->push($sub->id);
            // Recursive สำหรับหน่วยงานย่อยระดับลึกกว่า
            $deeperSubs = $this->getAllSubDepartmentIds($sub->id);
            $allIds = $allIds->merge($deeperSubs);
        }
        
    } catch (\Exception $e) {
        Log::error('Error getting sub-department IDs', [
            'faculty_id' => $facultyId,
            'error' => $e->getMessage()
        ]);
    }
    
    return $allIds;
}

/**
 * ปรับปรุง isInSameFaculty() ให้เข้มงวดมากขึ้น
 */
public function isInSameFaculty($targetDepartmentId): bool
{
    if (!$targetDepartmentId) {
        return false;
    }

    try {
        // หาคณะของผู้ใช้
        $userFacultyId = $this->getUserFacultyId();
        
        if (!$userFacultyId) {
            Log::info('User has no faculty for comparison', [
                'user_id' => $this->id,
                'target_department_id' => $targetDepartmentId
            ]);
            return false;
        }

        // หาคณะของ target department
        $targetFacultyId = $this->getDepartmentFacultyId($targetDepartmentId);
        
        if (!$targetFacultyId) {
            Log::info('Target department has no faculty', [
                'user_id' => $this->id,
                'target_department_id' => $targetDepartmentId
            ]);
            return false;
        }

        $isSameFaculty = $userFacultyId === $targetFacultyId;
        
        Log::debug('Faculty comparison result', [
            'user_id' => $this->id,
            'user_faculty_id' => $userFacultyId,
            'target_department_id' => $targetDepartmentId,
            'target_faculty_id' => $targetFacultyId,
            'is_same_faculty' => $isSameFaculty
        ]);

        return $isSameFaculty;

    } catch (\Exception $e) {
        Log::error('Error comparing faculties', [
            'user_id' => $this->id,
            'target_department_id' => $targetDepartmentId,
            'error' => $e->getMessage()
        ]);
        
        return false;
    }
}

    /**
     * ปรับปรุง getAccessibleDepartmentIds() ให้จำกัดตามคณะอย่างเข้มงวด
     */
    public function getAccessibleDepartmentIds(): \Illuminate\Support\Collection
    {
        // Super Admin เห็นทุกหน่วยงาน
        if ($this->isSuperAdmin()) {
            try {
                return Department::where('is_active', true)->pluck('id');
            } catch (\Exception $e) {
                Log::warning('Failed to get all departments for super admin', [
                    'user_id' => $this->id,
                    'error' => $e->getMessage()
                ]);
                return collect();
            }
        }

        // Admin/Manager - เข้าถึงได้ในคณะของตน
        if ($this->isAdmin() || $this->isManager()) {
            $manageableIds = $this->getManageableDepartmentIdsInFaculty();
            return collect($manageableIds);
        }

        // ผู้ใช้ทั่วไป - เข้าถึงได้เฉพาะในคณะเดียวกัน
        try {
            $userDepartment = $this->getUserDepartment();
            if (!$userDepartment) {
                Log::info('User has no department', ['user_id' => $this->id]);
                return collect();
            }

            $faculty = $this->getFacultyDepartment($userDepartment);
            if (!$faculty) {
                // ถ้าไม่มีคณะ ให้เข้าถึงได้เฉพาะหน่วยงานตัวเอง
                Log::info('User department has no faculty', [
                    'user_id' => $this->id,
                    'department_id' => $userDepartment->id
                ]);
                return collect([$userDepartment->id]);
            }

            // ดึงหน่วยงานทั้งหมดในคณะ
            $allIds = collect([$faculty->id]);
            
            if (method_exists($faculty, 'getAllDescendants')) {
                $subDepartments = $faculty->getAllDescendants();
                $subIds = $subDepartments->pluck('id');
                $allIds = $allIds->merge($subIds);
            } else {
                // Fallback method
                $allIds = $allIds->merge($this->getAllSubDepartmentIds($faculty->id));
            }
            
            $result = $allIds->unique()->values();
            
            Log::debug('Accessible departments calculated for regular user', [
                'user_id' => $this->id,
                'faculty_id' => $faculty->id,
                'accessible_count' => $result->count()
            ]);
            
            return $result;

        } catch (\Exception $e) {
            Log::warning('Failed to get accessible department IDs', [
                'user_id' => $this->id,
                'error' => $e->getMessage()
            ]);
            
            return collect();
        }
    }

    /**
     * ตรวจสอบว่าสามารถดูข้อมูลหน่วยงานได้หรือไม่ (Enhanced version)
     */
    public function canViewDepartmentData($departmentId): bool
    {
        if (!$departmentId) {
            return false;
        }

        // Super Admin ดูได้ทุกอย่าง
        if ($this->isSuperAdmin()) {
            return true;
        }

        // Admin ธรรมดา - ดูได้ในคณะของตน
        if ($this->isAdmin()) {
            return $this->isInSameFaculty($departmentId);
        }

        // ผู้ใช้ทั่วไป - ดูได้เฉพาะในคณะเดียวกัน
        $accessibleIds = $this->getAccessibleDepartmentIds();
        return $accessibleIds->contains($departmentId);
    }

    /**
     * เพิ่ม method สำหรับทดสอบการเข้าถึงระหว่าง users
     */
    public function canAccessSameFacultyAs(User $otherUser): bool
    {
        if ($this->isSuperAdmin() || $otherUser->isSuperAdmin()) {
            return true;
        }

        $thisFacultyId = $this->getUserFacultyId();
        $otherFacultyId = $otherUser->getUserFacultyId();

        if (!$thisFacultyId || !$otherFacultyId) {
            return false;
        }

        return $thisFacultyId === $otherFacultyId;
    }

    /**
     * ลบคำนำหน้าออกจากชื่อ
     */
    public static function removeNamePrefix(string $name): string
    {
        $name = trim($name);
        
        if (empty($name)) {
            return '';
        }
        
        $prefixes = self::getNamePrefixes();
        
        // เรียงจากยาวไปสั้น เพื่อจับคำนำหน้าที่ยาวก่อน
        usort($prefixes, function($a, $b) {
            return strlen($b) - strlen($a);
        });
        
        // ลองลบ prefix ซ้ำจนกว่าจะไม่มี prefix ให้ลบแล้ว
        $maxIterations = 5; // ป้องกัน infinite loop
        $iteration = 0;
        
        while ($iteration < $maxIterations) {
            $originalName = $name;
            
            foreach ($prefixes as $prefix) {
                if (str_starts_with($name, $prefix)) {
                    $nameWithoutPrefix = trim(substr($name, strlen($prefix)));
                    
                    // ถ้าลบ prefix แล้วไม่มีอะไรเหลือ ให้คืนค่าเดิม
                    if (empty($nameWithoutPrefix)) {
                        return $name; // คืนค่าชื่อเดิมที่มี prefix
                    }
                    
                    $name = $nameWithoutPrefix;
                    break; // ออกจาก foreach และลองรอบใหม่
                }
            }
            
            // ถ้าไม่มีการเปลี่ยนแปลง แสดงว่าไม่มี prefix ให้ลบแล้ว
            if ($name === $originalName) {
                break;
            }
            
            $iteration++;
        }
        
        return $name;
    }

    /**
     * เปรียบเทียบชื่อโดยไม่สนใจคำนำหน้า
     */
    public static function namesMatch(string $name1, string $name2): bool
    {
        $cleanName1 = self::removeNamePrefix(trim($name1));
        $cleanName2 = self::removeNamePrefix(trim($name2));
        
        // เปรียบเทียบแบบ case-insensitive และลบช่องว่างเพิ่มเติม
        $normalizedName1 = preg_replace('/\s+/', ' ', strtolower($cleanName1));
        $normalizedName2 = preg_replace('/\s+/', ' ', strtolower($cleanName2));
        
        return $normalizedName1 === $normalizedName2;
    }

    /**
     * ค้นหาผู้ใช้โดยจับคู่ชื่อ (ไม่สนใจคำนำหน้า) และหน่วยงาน
     * พร้อมตรวจสอบอีเมลเป็นตัวเลือกเสริม
     */
    public static function findUserByNameAndDepartment(string $name, string $department, ?string $email = null): ?User
    {
        Log::info('Searching for user by name and department', [
            'search_name' => $name,
            'search_department' => $department,
            'search_email' => $email
        ]);

        // ลบคำนำหน้าจากชื่อที่ต้องการค้นหา
        $cleanSearchName = self::removeNamePrefix($name);
        
        try {
            // หา department_id
            $departmentRecord = Department::where('name', $department)->first();
            
            if (!$departmentRecord) {
                Log::warning('Department not found', ['department' => $department]);
                
                // ลองหาด้วย code หรือคล้ายกัน
                $departmentRecord = Department::where('code', $department)
                    ->orWhere('name', 'LIKE', "%{$department}%")
                    ->first();
                
                if (!$departmentRecord) {
                    Log::warning('Department still not found after fuzzy search', ['department' => $department]);
                    // ไม่ return null ทันที แต่ลองหาโดยไม่ระบุหน่วยงาน
                }
            }

            // ค้นหาผู้ใช้ในหน่วยงานนั้น (หากหาหน่วยงานเจอ)
            $users = collect();
            
            if ($departmentRecord) {
                // ค้นหาผู้ใช้ที่มี relation กับหน่วยงาน
                $usersFromPivot = User::whereHas('departments', function($query) use ($departmentRecord) {
                    $query->where('departments.id', $departmentRecord->id);
                })->get();
                
                // ค้นหาผู้ใช้ที่มี department field ตรงกัน
                $usersFromField = User::where('department', $department)->get();
                
                $users = $usersFromPivot->merge($usersFromField)->unique('id');
                
                Log::info('Found users in department', [
                    'department_id' => $departmentRecord->id,
                    'users_from_pivot' => $usersFromPivot->count(),
                    'users_from_field' => $usersFromField->count(),
                    'total_unique' => $users->count()
                ]);
            } else {
                // หากไม่เจอหน่วยงาน ให้ค้นหาจากผู้ใช้ทั้งหมดที่มี department field ตรงกัน
                $users = User::where('department', $department)->get();
                
                Log::info('Searching without department record', [
                    'users_found' => $users->count()
                ]);
            }

            // ตรวจสอบการจับคู่ชื่อ
            foreach ($users as $user) {
                if (self::namesMatch($user->name, $name)) {
                    Log::info('Name match found', [
                        'existing_user_id' => $user->id,
                        'existing_name' => $user->name,
                        'search_name' => $name,
                        'clean_existing' => self::removeNamePrefix($user->name),
                        'clean_search' => $cleanSearchName
                    ]);
                    return $user;
                }
            }

            // ถ้ามี email ให้ลองหาด้วย email เป็นตัวเลือกสุดท้าย
            if ($email) {
                $userByEmail = User::where('email', $email)->first();
                if ($userByEmail) {
                    Log::info('User found by email fallback', [
                        'user_id' => $userByEmail->id,
                        'email' => $email
                    ]);
                    return $userByEmail;
                }
            }

        } catch (\Exception $e) {
            Log::error('Error in findUserByNameAndDepartment', [
                'error' => $e->getMessage(),
                'file' => $e->getFile(),
                'line' => $e->getLine(),
                'search_name' => $name,
                'search_department' => $department
            ]);
            
            // ในกรณีที่เกิด error ให้ลองหาแบบง่าย ๆ
            try {
                $simpleUsers = User::where('department', $department)->get();
                foreach ($simpleUsers as $user) {
                    if (self::namesMatch($user->name, $name)) {
                        Log::info('Name match found via fallback method', [
                            'user_id' => $user->id,
                            'existing_name' => $user->name
                        ]);
                        return $user;
                    }
                }
            } catch (\Exception $fallbackError) {
                Log::error('Fallback search also failed', [
                    'error' => $fallbackError->getMessage()
                ]);
            }
        }

        Log::info('No matching user found');
        return null;
    }

    /**
     * ค้นหาผู้ใช้ด้วยหลายเกณฑ์ (สำหรับ KU ALL LOGIN)
     * ลำดับความสำคัญ: email -> name+department -> name เท่านั้น -> similar email
     */
    public static function findExistingKUUser(string $email, string $name, string $department): ?User
    {
        Log::info('Searching for existing KU user', [
            'email' => $email,
            'name' => $name,
            'department' => $department
        ]);

        // 1. ลองหาด้วย email ก่อน (วิธีหลัก)
        $userByEmail = User::where('email', $email)->first();
        if ($userByEmail) {
            Log::info('User found by email', ['user_id' => $userByEmail->id]);
            return $userByEmail;
        }

        // 2. ลองหาด้วยชื่อ + หน่วยงาน
        $userByNameDept = self::findUserByNameAndDepartment($name, $department);
        if ($userByNameDept) {
            Log::info('User found by name and department', ['user_id' => $userByNameDept->id]);
            return $userByNameDept;
        }

        // 3. ลองหาด้วยชื่อเท่านั้น (ในกรณีที่มีการเปลี่ยนหน่วยงาน)
        // ✅ ปรับปรุง: ใช้ chunk เพื่อความประหยัดหน่วยความจำ
        $foundUser = null;
        User::chunk(200, function ($users) use ($name, &$foundUser) {
            foreach ($users as $user) {
                if (self::namesMatch($user->name, $name)) {
                    $foundUser = $user;
                    return false; // หยุด chunk ทันทีที่เจอ
                }
            }
        });

        if ($foundUser) {
            Log::info('User found by name only (different department)', [
                'user_id' => $foundUser->id,
                'existing_department' => $foundUser->department
            ]);
            return $foundUser;
        }

        // 4. 🔥 ลองหาด้วย username ที่คล้ายกัน (สำหรับกรณี email domain ต่างกัน)
        // เช่น kantarasukan.s@ku.th vs fagikkso@ku.ac.th
        $emailPrefix = explode('@', $email)[0]; // ได้ fagikkso
        $similarUsers = User::where('email', 'LIKE', '%' . $emailPrefix . '%')
                           ->orWhere('email', 'LIKE', '%' . str_replace('.', '', $emailPrefix) . '%')
                           ->get();
                           
        foreach ($similarUsers as $user) {
            if (self::namesMatch($user->name, $name)) {
                Log::info('User found by similar email pattern and name match', [
                    'user_id' => $user->id,
                    'existing_email' => $user->email,
                    'ku_email' => $email,
                    'name_match' => true
                ]);
                return $user;
            }
        }

        // 5. 🔥 ลองหาด้วยการเปรียบเทียบ username ใน email
        // กรณีที่ KU ให้ username ใหม่แต่ชื่อจริงเดียวกัน
        $existingUsers = User::where('department', $department)
                            ->orWhereHas('departments', function($query) use ($department) {
                                $query->where('name', $department);
                            })
                            ->get();
                            
        foreach ($existingUsers as $user) {
            if (self::namesMatch($user->name, $name)) {
                // เช่น kantarasukan.s@ku.th และ fagikkso@ku.ac.th แต่ชื่อ "กัณฑลสุกานต์ สมบูรณ์" ตรงกัน
                Log::info('User found by name in same department (different username)', [
                    'user_id' => $user->id,
                    'existing_email' => $user->email,
                    'ku_email' => $email,
                    'existing_name' => $user->name,
                    'ku_name' => $name
                ]);
                return $user;
            }
        }

        Log::info('No existing user found');
        return null;
    }

    /**
     * 🔥 Fixed version of accessible departments for filter
     * Returns Collection of Department models
     * Stricter logic: Non-Admins only see explicitly assigned departments
     */
    public function getAccessibleDepartmentsForFilterFixed()
    {
        // 1. Super Admin sees all
        if ($this->isSuperAdmin()) {
            return Department::where('is_active', true)
                ->orderBy('type', 'desc')
                ->orderBy('name')
                ->get();
        }

        // 2. Admin/Manager: Check if they manage a MAIN department
        $managesMainDept = $this->departmentUsers()
            ->where('is_active', true)
            ->whereIn('role', ['admin', 'manager'])
            ->whereHas('department', function ($q) {
                $q->where('type', 'main');
            })
            ->exists();

        if ($managesMainDept) {
            // If they manage a main department, they see EVERYTHING in that faculty
            $deptIds = $this->getManageableDepartmentIdsInFaculty();
            return Department::whereIn('id', $deptIds)
                ->where('is_active', true)
                ->orderBy('name')
                ->get();
        }

        // 3. Project Manager / Staff / General User
        // Only see departments they are actively assigned to
        // Do NOT expand to entire faculty
        $deptIds = $this->activeDepartmentUsers()
            ->pluck('department_id')
            ->unique()
            ->toArray();

        // Also include sub_department_id (e.g. สถานีวิจัยดอยปุย)
        $subDeptIds = $this->activeDepartmentUsers()
            ->whereNotNull('sub_department_id')
            ->pluck('sub_department_id')
            ->unique()
            ->toArray();

        $deptIds = array_unique(array_merge($deptIds, $subDeptIds));
            
        // Fallback to single department field if no pivot records
        if (empty($deptIds) && !empty($this->department)) {
            $dept = Department::where('name', $this->department)->first();
            if ($dept) {
                $deptIds[] = $dept->id;
            }
        }

        // Also include the department they created (if any)
        $createdDept = Department::where('created_by', $this->id)->first();
        if ($createdDept) {
            $deptIds[] = $createdDept->id;
        }

        return Department::whereIn('id', array_unique($deptIds))
            ->where('is_active', true)
            ->orderBy('name')
            ->get();
    }
